\item \labelcref{it:metric-subspace-is-ms}: metric subspace is a metric space
\item \labelcref{it:openness-equiv-def}: equivalent definition of open set

\item \labelcref{it:open-ball-open}: open ball in \(X\) is open in \(X\)
\item \labelcref{it:closed-ball-closed}: closed ball in \(X\) is closed in \(X\)
\item \Cref{prp:open-set-exist-open-ball}: criterion of openness (involving open ball with unknown center)
\item \labelcref{it:open-sets-union-open}: arbitrary union of open sets in \(X\) is open in \(X\)
\item \labelcref{it:open-sets-intersect-open}: finite intersection of open sets in \(X\) is open in \(X\)
\item \labelcref{it:closed-sets-union-closed}: finite union of closed sets in \(X\) is closed in \(X\)

\item \Cref{thm:union-disjoint-int}: open subset of \(\R\) can be uniquely
expressed as a union of countably many disjoint open intervals
\item \Cref{thm:closure-smallest-closed}: closure of \(S\) is the smallest closed subset of \(X\) which contains all points in \(S\)
\item \Cref{prp:adherent-zero-dist-from-set}: criterion for adherent point based on distance between point and set

\item \labelcref{it:closure-decompose-bound-interior}: closure is a union of set of all boundary points and set of all interior points
\item \labelcref{it:int-clos-deriv-bound-prop-summary}: summary of properties of interiors, closures, derived sets, and boundaries

\item \Cref{thm:cpt-bw-prop}: compactness implies Boltzano-Weierstrass property

\item \labelcref{it:cpt-union-cpt}: finite union of compact sets in \(X\) is compact in \(X\)
\item \labelcref{it:cpt-intersect-cpt}: arbitrary intersection of compact sets in \(X\) is compact in \(X\)
\item \Cref{thm:boltzano-weierstrass}: Boltzano-Weierstrass theorem
\item \Cref{thm:cantor-intersection}: Cantor intersection theorem
\item \Cref{thm:heine-borel}: Heine-Borel theorem (closedness and boundedness

under the metric space \(\R^n\)
